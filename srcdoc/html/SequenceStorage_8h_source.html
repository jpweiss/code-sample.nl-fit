170  static void assign(ref_vec_t targ, constref_vec_t src, size_t src_size)
171  {
172  if(targ == src) {
173  // Ignore self-assignment attempts.
174  return;
175  }
176  vec_t copy_of_src(new_copy_of(src, src_size));
177  vec_t oldtarg(targ);
178  targ = copy_of_src;
179  destroy(oldtarg);
180  }

188  static bool equivalent(constref_vec_t a, constref_vec_t b,
189  size_t size_both)
190  {
191  if(a == b) {
192  // Trivial case: self-comparison.
193  return true;
194  }
195 
196  for(size_t i=0; i<size_both; ++i) {

198  return false;
